35  private NewMethodGenerator(String modifier, String name,
36  Class<?>[] paramTypes, Class<?> returnType,
37  Class<? extends Throwable>[] exceptionTypes) {
38  StringBuffer sb = new StringBuffer();
39  sb.append(modifier);
40  sb.append(" ");
41  sb.append(ClassHelper.makeClassName(returnType));
42  sb.append(" ");
43  sb.append(name);
44  sb.append("(");
45  if (paramTypes != null && paramTypes.length > 0) {
46  int i = 0;
47  for (Class<?> paramType : paramTypes) {
48  if (i != 0) {
49  sb.append(",");
50  }
51  sb.append(ClassHelper.makeClassName(paramType));
52  }
53  }
54  sb.append(")");
55  if (exceptionTypes != null && exceptionTypes.length > 0) {
56  sb.append("throws ");
57  int i = 0;
58  for (Class<? extends Throwable> exceptionType : exceptionTypes) {
59  if (i != 0) {
60  sb.append(",");
61  }
62  sb.append(ClassHelper.makeClassName(exceptionType));
63  }
64  }
65  this.signature = sb.toString();
66  }

68  private String getMethodCode() {
69  StringBuffer sb = new StringBuffer();
70  sb.append(signature);
71  sb.append('{');
72  for (String line : methodLines) {

74  }
75  sb.append('}');

126  public Field addLocalVariableDefine(Class<?> type, Expression expr) {
127  String name = generateMethodVariableName();
128  StringBuilder sb = new StringBuilder();
129  sb.append(ClassHelper.makeClassName(type)).append(' ');
130  sb.append(name);
131  if (expr != null) {
132  sb.append('=');
133  sb.append(expr.cast(type).getCode());
134  }
135  sb.append(';');
136  methodLines.add(sb.toString());
