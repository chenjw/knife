38  private String lineListToCode(List<String> lines) {
39  if (lines == null || lines.size() == 0) {
40  return null;
41  } else {
42  StringBuffer sb = new StringBuffer();

45  }
46  return sb.toString();
47  }

51  public void generate(CtClass ctClass) throws CannotCompileException,
52  NotFoundException {
53  String beforeCode = lineListToCode(beforeMethodLines);
54  if (beforeCode != null) {
55 
56  try {
57  oldCtMethod.insertBefore(beforeCode);
58  } catch (CannotCompileException e) {
59  throw e;
60  }
61 
62  }
63  String afterCode = lineListToCode(afterMethodLines);
64  if (afterCode != null) {
65  try {
66  oldCtMethod.insertAfter(afterCode, false);
67  } catch (CannotCompileException e) {
68  throw e;
69  }
70 
71  }
72  String catchCode = lineListToCode(catchMethodLines);
73  if (catchCode != null) {
74  oldCtMethod.addCatch(catchCode + "throw $e;",
75  classGenerator.findCtClass("java.lang.Throwable"));
76  }
77  String finallyCode = lineListToCode(finallyMethodLines);
78  if (finallyCode != null) {
79  oldCtMethod.insertAfter(finallyCode, true);
80  }
81  }

126  public Field addLocalVariableDefine(TypeEnum linetype, Class<?> type,
127  Expression expr) {
128  String name = generateMethodVariableName();

130  sb.append(ClassHelper.makeClassName(type)).append(' ');
