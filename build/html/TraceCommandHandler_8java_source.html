61  private void configStrategy(Args args, MethodInfo methodInfo)

130  private MethodInfo findMethod(Args args) throws Exception {
131  MethodInfo methodInfo = new MethodInfo();
132  String m = args.arg("trace-expression");
133  m = m.trim();
134  Method method = null;
135  if (StringHelper.isNumeric(m)) {
136  method = ((Method[]) ServiceRegistry.getService(
137  ContextService.class).get(Constants.METHOD_LIST))[Integer
138  .parseInt(m)];
139 
140  } else {
141  if (m.indexOf(".") != -1) {
142  String className = StringHelper.substringBeforeLast(m, ".");
143  m = StringHelper.substringAfterLast(m, ".");
144  Class<?> clazz = NativeHelper.findLoadedClass(className);
145  if (clazz == null) {
146  clazz = Helper.findClass(className);
147  }
148  if (clazz == null) {
149 
150  Agent.sendResult(ResultHelper.newErrorResult("class "
151  + className + " not found!"));
152  return null;
153  }
154  Method[] methods = ReflectHelper.getMethods(clazz);
155  for (Method tm : methods) {
156  if (StringHelper.equals(tm.getName(), m)) {
157  if (Modifier.isStatic(tm.getModifiers())) {
158  method = tm;
159  break;
160  }
161  }
162  }
163  } else {
164  Object obj = ServiceRegistry.getService(ContextService.class)
165  .get(Constants.THIS);
166  if (obj == null) {
167  Agent.sendResult(ResultHelper.newErrorResult("not found!"));
168  return null;
169  }
170  Method[] methods = obj.getClass().getMethods();
171  for (Method tm : methods) {
172  if (StringHelper.equals(tm.getName(), m)) {
173  method = tm;
174  break;
175  }
176  }
177  }
178  }
179  if (method == null) {
180  Agent.sendResult(ResultHelper.newErrorResult("cant find method!"));
181  return null;
182  }
183  methodInfo.setMethod(method);
184  if (Modifier.isStatic(method.getModifiers())) {
185  methodInfo.setClazz(method.getDeclaringClass());
186  methodInfo.setThisObject(null);
187  } else {
188  Object thisObject = ServiceRegistry
189  .getService(ContextService.class).get(Constants.THIS);
190  methodInfo.setThisObject(thisObject);
191  methodInfo.setClazz(thisObject.getClass());
192  }
193  return methodInfo;
194  }
