85  if (fOptions != null) {

87  .get("filter-expression")));

89  filters.add(new ProxyMethodFilter());
90  filters.add(new TraceMethodFilter(methodInfo.getThisObject(),
91  methodInfo.getClazz(), methodInfo.getMethod()));
92  filters.add(new TraceTimesCountFilter(traceNum));
93  filters.add(new EnterLeavePrintFilter());
94  filters.add(new DepthFilter());
95  if (tOptions == null && cOptions==null) {
96  filters.add(new Depth0Filter());
97  }
98  filters.add(new TimingFilter());
99  if(cOptions!=null){
100  filters.add(new StatPrintFilter());

104  }
