66  public HeapHistogramImpl(String histogramText) {
67  Map<String, ClassInfoImpl> classesMap = new HashMap(1024);
68  Map<String, ClassInfoImpl> permGenMap = new HashMap(1024);
69  time = new Date();
70  Scanner sc = new Scanner(histogramText);
71  sc.useRadix(10);
72  while (!sc.hasNext("-+")) {
73  sc.nextLine();
74  }
75  sc.skip("-+");
76  sc.nextLine();
77 
78 
79  while (sc.hasNext("[0-9]+:")) { // NOI18N
80  ClassInfoImpl newClInfo = new ClassInfoImpl(sc);
81  storeClassInfo(newClInfo, classesMap);
82  totalHeapBytes += newClInfo.getBytes();
83  totalHeapInstances += newClInfo.getInstancesCount();
84  }
85  sc.next("Total"); // NOI18N
86  totalInstances = sc.nextLong();
87  totalBytes = sc.nextLong();
88  classes = new HashSet(classesMap.values());
89  }

142  private String convertJVMName(String jvmName) {
143  String className = null;
144  int index = jvmName.lastIndexOf('['); // NOI18N
145 
146  if (index != -1) {
147  switch (jvmName.charAt(index + 1)) {
148  case BOOLEAN_CODE:
149  className = BOOLEAN_TEXT;
150  break;
151  case CHAR_CODE:
152  className = CHAR_TEXT;
153  break;
154  case BYTE_CODE:
155  className = BYTE_TEXT;
156  break;
157  case SHORT_CODE:
158  className = SHORT_TEXT;
159  break;
160  case INT_CODE:
161  className = INT_TEXT;
162  break;
163  case LONG_CODE:
164  className = LONG_TEXT;
165  break;
166  case FLOAT_CODE:
167  className = FLOAT_TEXT;
168  break;
169  case DOUBLE_CODE:
170  className = DOUBLE_TEXT;
171  break;
172  case OBJECT_CODE:
173  className = jvmName.substring(index + 2, jvmName.length() - 1);
174  break;
175  default:
176  System.err.println("Uknown name " + jvmName); // NOI18N
177  className = jvmName;
178  }
179  for (int i = 0; i <= index; i++) {
180  className += "[]";
181  }
182  }
183  if (className == null) {
184  className = jvmName;
185  }
186  return className.intern();
187  }
188  }
