26  private void addThreadInfo(PerfTree perfTree, ThreadInfo[] tis) {
27  for (ThreadInfo ts : tis) {
28  PerfNode perfNode = perfTree.getThreadRoots().get(ts.getThreadName());
29  if (perfNode == null) {
30  perfNode = new PerfNode();
31  perfTree.getThreadRoots().put(ts.getThreadName(), perfNode);
32  }
33  perfNode.setTimecost(perfNode.getTimecost() + perfInterval);
34  if (ts.getThreadState() == Thread.State.RUNNABLE) {
35  perfNode.setTimecostCpu(perfNode.getTimecostCpu() + perfInterval);
36  }
37  StackTraceElement[] stes = ts.getStackTrace();
38  if (stes == null) {
39  continue;
40  }
41  PerfNode currentNode = perfNode;
42  for (int i = 0; i < stes.length; i++) {
43  StackTraceElement ste = stes[stes.length - i - 1];
44  PerfNodeKey key = new PerfNodeKey();
45  key.setClassName(ste.getClassName());
46  key.setMethodName(ste.getMethodName());
47  PerfNode node = currentNode.getChildren().get(key);
48  if (node == null) {
49  node = new PerfNode();
50  currentNode.getChildren().put(key, node);
51  }
52  node.setTimecost(node.getTimecost() + perfInterval);
53  if (ts.getThreadState() == Thread.State.RUNNABLE) {
54  node.setTimecostCpu(node.getTimecostCpu() + perfInterval);
55  }
56  // 当前节点是最后节点，说明时间都是被这个节点占用的
57  if (i == stes.length - 1) {
58  node.setSelfTimecost(node.getSelfTimecost() + perfInterval);
59  if (ts.getThreadState() == Thread.State.RUNNABLE) {
60  node.setSelfTimecostCpu(node.getSelfTimecostCpu() + perfInterval);
61  }
62  }
63  currentNode = node;
64  }
65  }
