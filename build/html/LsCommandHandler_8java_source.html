47  private void lsField(Args args) {
48  ClassOrObject target = findTarget(args);

50  Agent.sendResult(ResultHelper.newErrorResult("not found!"));

53  List<FieldInfo> fieldInfos = new ArrayList<FieldInfo>();
54  if (target.getClazz() != null) {
55  Map<Field, Object> fieldMap = NativeHelper
56  .getStaticFieldValues(target.getClazz());
57  for (Entry<Field, Object> entry : fieldMap.entrySet()) {
58  FieldInfo info = new FieldInfo();
59  info.setStatic(true);
60  info.setName(entry.getKey().getName());
61  if (entry.getValue() != null) {
62  ObjectInfo fValue = new ObjectInfo();
63  fValue.setObjectId(ServiceRegistry.getService(
64  ObjectHolderService.class).toId(entry.getValue()));
65  fValue.setValueString(toString(args, entry.getValue()));
66  info.setValue(fValue);
67  }
68  fieldInfos.add(info);
69  }
70  }
71  if (target.getObj() != null) {
72  Map<Field, Object> fieldMap = NativeHelper.getFieldValues(target
73  .getObj());
74  for (Entry<Field, Object> entry : fieldMap.entrySet()) {
75  FieldInfo info = new FieldInfo();
76  info.setStatic(false);
77  info.setName(entry.getKey().getName());
78  if (entry.getValue() != null) {
79  ObjectInfo fValue = new ObjectInfo();
80  fValue.setObjectId(ServiceRegistry.getService(
81  ObjectHolderService.class).toId(entry.getValue()));
82  fValue.setValueString(toString(args, entry.getValue()));
83  info.setValue(fValue);
