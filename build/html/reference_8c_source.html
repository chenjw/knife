26 void addObjectReferrer(ObjectReferrerList * list, jlong count) {
27  // 从根元素开始依次遍历，从小到大排，找到合适的位置
28  ObjectReferrer ** pBefore = &list->root;
29  ObjectReferrer * after = 0;
30  while (1) {
31  ObjectReferrer * current = *pBefore;
32  if (current == 0 ) {
33  break;
34  }
35  if (count < current->count) {
36  pBefore = &(current->next);
37  } else {
38  after = current;
39  break;
40  }
41  }
42  // 如果链表超过最大长度就舍弃掉根节点（最小）的元素。
43  if (list->num >= list->maxNum) {
44 
45  if (pBefore == &list->root) {
46  return;
47  } else {
48  ObjectReferrer * referrer = 0;
49  referrer = (ObjectReferrer *) allocate(sizeof(ObjectReferrer));
50  referrer->count = count;
51  referrer->next = after;
52  *pBefore = referrer;
54  list->root = list->root->next;
55  }
56  }
57  // 如果没有超过最大长度就接到根节点的位置
58  else {
59  list->num++;
60  ObjectReferrer * referrer = 0;
61  referrer = (ObjectReferrer *) allocate(sizeof(ObjectReferrer));
62  referrer->count = count;
63  referrer->next = after;
64  *pBefore = referrer;
65  }
66 }
