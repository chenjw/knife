129  private void invoke(boolean isTrace, Method method, Object thisObject,
130  Object[] args) throws Exception {
131  // 重置classloader
132  ClassLoaderHelper.resetClassLoader(method.getDeclaringClass());
133  boolean isStatic = Modifier.isStatic(method.getModifiers());
134  Class<?> clazz = null;
135  if (isStatic) {
136  thisObject = null;
137  clazz = method.getDeclaringClass();
138  } else {
139  clazz = thisObject.getClass();
140  }
141  try {
142  Profiler.start(thisObject, clazz.getName(), method.getName(), args,
143  null, -1);
144  if (isTrace) {
145  Profiler.profile(thisObject, method);
146  }
147  InvokeResult r = null;
148  if (isStatic) {
149  r = ReflectHelper.invokeStaticMethod(method, args);
150  } else {
151  r = ReflectHelper.invokeMethod(thisObject, method, args);
152  }
153 
154  if (r.isSuccess()) {
155  Profiler.returnEnd(thisObject, clazz.getName(),
156  method.getName(), args, r.getResult());
157  } else {
158  Profiler.exceptionEnd(thisObject, clazz.getName(),
159  method.getName(), args, r.getE());
160  }
161  } catch (MethodInvokeException e) {
162  Profiler.exceptionEnd(thisObject, clazz.getName(),
163  method.getName(), args, e);
164  throw e;
165  }
166  }
