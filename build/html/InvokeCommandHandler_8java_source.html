135  private void invoke(boolean isTrace, Method method, Object thisObject,
136  Object[] args) throws Exception {
137  // 重置classloader
138  ClassLoaderHelper.resetClassLoader(method.getDeclaringClass());
139  boolean isStatic = Modifier.isStatic(method.getModifiers());
140  Class<?> clazz = null;
141  if (isStatic) {
142  thisObject = null;
143  clazz = method.getDeclaringClass();
144  } else {
145  clazz = thisObject.getClass();
146  }
147  try {
148  Profiler.start(thisObject, clazz.getName(), method.getName(), args,
149  null, "-1");
150  if (isTrace) {
151  Profiler.profile(thisObject, method);
152  }
153  InvokeResult r = null;
154  if (isStatic) {
155  r = ReflectHelper.invokeStaticMethod(method, args);
156  } else {
157  r = ReflectHelper.invokeMethod(thisObject, method, args);
158  }
159  if (r.isSuccess()) {
160  Profiler.returnEnd(thisObject, clazz.getName(),
161  method.getName(), args, r.getResult());
162  } else {
163  Profiler.exceptionEnd(thisObject, clazz.getName(),
164  method.getName(), args, r.getE());
165  }
166  } catch (MethodInvokeException e) {
167  Profiler.exceptionEnd(thisObject, clazz.getName(),
168  method.getName(), args, e);
169  throw e;
170  }
171  }
